<Steps>

    ![](/images/tutorials/dashboard2-e.png)

    ![](/images/tutorials/assignments-selections2-e.png)
  </Step>

  <Step title="Click &#x22;Edit&#x22; tab link.">
    ![](/images/tutorials/assignments-edit1-e.png)
  </Step>

  <Step title="Click &#x22;Delete&#x22; button. A pop-up will appear.">
    ![](/images/tutorials/assignments-delete1-e.png)
  </Step>

  <Step title="Click the &#x22;Delete&#x22; button for confirmation.">
    ![](/images/tutorials/assignments-delete2-e.png)
  </Step>
